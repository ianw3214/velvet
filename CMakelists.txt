cmake_minimum_required(VERSION 3.20)
project(Maple)

find_package(LLVM REQUIRED CONFIG)
find_package(Catch2 CONFIG REQUIRED)

message(STATUS "Found LLVM ${LLVM_PACKAGE_VERSION}")
message(STATUS "Using LLVMConfig.cmake in: ${LLVM_DIR}")

include_directories(${LLVM_INCLUDE_DIRS})
separate_arguments(LLVM_DEFINITIONS_LIST NATIVE_COMMAND ${LLVM_DEFINITIONS})
add_definitions(${LLVM_DEFINITIONS_LIST})

add_library(Maple)
target_include_directories(Maple PRIVATE source)

add_executable(MapleMain)
add_subdirectory(source)
target_link_libraries(MapleMain Maple)
set_property(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY VS_STARTUP_PROJECT MapleMain)

add_executable(MapleTests)
target_include_directories(MapleTests PRIVATE source)
add_subdirectory(tests)
target_link_libraries(MapleTests Maple)

# Find the libraries that correspond to the LLVM components
# that we wish to use
llvm_map_components_to_libnames(llvm_libs support core irreader)

# Link against LLVM libraries
target_link_libraries(Maple ${llvm_libs})

target_link_libraries(MapleTests ${llvm_libs})
target_link_libraries(MapleTests Catch2::Catch2)
